Sample Code:

        
            let stack = [];
            // Add items to the stack
            stack.push('A');
            stack.push('B');
            stack.push('C');
    
            // Remove items from the stack
            let item1 = stack.pop();
            let item2 = stack.pop();
            let item3 = stack.pop();
    
            console.log(item1); // 'C'
            console.log(item2); // 'B'
            console.log(item3); // 'A'
